\section{Problema e objetivo}
% ======================================================================
\begin{frame}{O problema P2}
\textbf{Posicionamento de relés em RSSF móvel.}
\begin{itemize}
  \item Região $\Omega\subset\mathbb{R}^2$, sorvedouro $s$, $N$ candidatos $Q$,
        $M$ sensores móveis, raio de alcance $R$.
  \item \textbf{P2:} menor $P\subseteq Q$ tal que, em todo instante $t$, cada móvel
        alcance $s$ por uma cadeia de relés instalados dentro do alcance.
  \item \textbf{Cromossomo:} $x\in\{0,1\}^N$, $x_j=1$ se o candidato $j$ é instalado.
  \item \textbf{Aptidão substituta} $F(x)$: conectividade $-$ penalidades (relés,
        saltos, distância, redundância). Maior é melhor.
\end{itemize}
\vfill
\begin{block}{Pergunta de dimensionamento}
Dado $1-\alpha$, qual o \emph{menor} tamanho de população $\Nhat$ para o AG achar
uma boa solução com probabilidade $\ge 1-\alpha$?
\end{block}
\end{frame}

% ======================================================================
\section{A fórmula de ruína do jogador}
% ======================================================================
\begin{frame}{O estimador: ruína do jogador (Harik et al.)}
\[
\boxed{\;\widehat{n}_u \;=\; -\ln(\alpha)\,2^{k_u-1}\,
\frac{\widehat{\sigma}_{BB,u}\sqrt{2m}}{\widehat{d}_u}\;}
\qquad \Nhat=\max_u\lceil \widehat n_u\rceil
\]
\vfill
\begin{columns}[T]
\column{0.5\textwidth}
\small
\begin{tabular}{ll}
\toprule
$\alpha$ & prob.\ de perder o bloco ($0{,}05$)\\
$k_u$ & ordem do bloco\\
$m$ & nº de blocos\\
$\widehat\sigma_{BB,u}$ & ruído do bloco\\
$\widehat d_u$ & sinal (vantagem do bloco correto)\\
\bottomrule
\end{tabular}
\column{0.5\textwidth}
\small
\textbf{Sinal e ruído} por diferenças entre amostras:
\[
\Delta_u^{(r)} = F(H_u^*,x_{-u}^{(r)})-F(H_u^L,x_{-u}^{(r)})
\]
\[
\widehat d_u=\overline{\Delta_u},\quad
\widehat\sigma_{BB,u}=\mathrm{sd}(\Delta_u)
\]
com $x_{-u}\sim\mathrm{Bernoulli}(0{,}5)$, $R=60$.
\end{columns}
\end{frame}

\begin{frame}{O ponto crítico: a \emph{ordem} do bloco}
\begin{itemize}
  \item O fator $2^{k-1}$ \alert{dobra a cada bit} do bloco.
  \item Blocos de grade arbitrária ($k$ até 9) $\Rightarrow$ $\Nhat=\mathbf{2839}$.
  \item Componentes conexas ($k=12$) $\Rightarrow$ $\sim\!6\times10^4$.
\end{itemize}
\vfill
\begin{alertblock}{P2 não tem engano do tipo Trap-$k$}
Uma região de relés admite \emph{muitas} coberturas aceitáveis. Sua dificuldade são
\textbf{muitos genes indispensáveis isolados}, não um bloco grande.
$\Rightarrow$ \textbf{blocos de ordem 1} ($k_u=1$, logo $2^{0}=1$).
\end{alertblock}
\vfill
\centering
Cada candidato indispensável $u$ é um bloco $\{u\}$: $H_u^*$ = ligado,
$H_u^L$ = desligado. Mantém-se $u$ só se o sinal é significativo
($\widehat d_u>1{,}645\,\widehat\sigma_{BB,u}/\sqrt R$).
\end{frame}

\section{Os cinco métodos como fontes de BB}
% ======================================================================
\begin{frame}{Uma fórmula, cinco fontes de blocos construtivos}
\centering
A fórmula é o \textbf{único estimador}. Cada método apenas \textbf{direciona os BBs}:
qual conjunto de genes é bloco e qual é o alelo correto $H_u^*$.
\vfill
\footnotesize
\begin{tabular}{lll}
\toprule
Método & Sinal / visão & Regra de BB (gene\_set)\\
\midrule
\textbf{M1} inferência & dinâmica da busca do AG & genes nas elites do AG\\
\textbf{M2} adjacência & alcançabilidade geométrica & $\cov(u)\ge\theta TM$\\
\textbf{M3} roteamento & uso operacional (caminhos) & $\route(u)\ge\phi$\\
\textbf{M4} MILP & ótimo exato & genes do ótimo MILP\\
\textbf{M5} união & evidências combinadas & $\cov\ge\theta TM \ \vee\ \route\ge\phi$\\
\bottomrule
\end{tabular}
\vfill
\normalsize
Sinal/ruído e fórmula \emph{idênticos}; muda só \emph{quais} genes cada fonte
aponta. Núcleo único: \texttt{methods/bb\_core.py}.
\end{frame}

% ---------------------------------------------------------------- M1
\begin{frame}{M1 --- Inferência: BBs pela dinâmica do AG}
\begin{columns}[T]
\column{0.56\textwidth}
\small
\textbf{Sinal:} a qualidade convergida satura,
$F^*(n)=F_\infty-A\,e^{-n/\tau}$ ($\tau\approx7{,}2$); a busca \emph{decide}
quais genes manter.
\medskip

\textbf{Regra de BB:} genes ligados em $\ge\!80\%$ das \emph{elites} (top 5\% por
$F$ do fluxo de indivíduos avaliados pelo AG).
\medskip

\textbf{Interpretação:} genes em que \emph{toda} boa solução concorda = os blocos
para os quais a busca convergiu.
\medskip

\textbf{$H_u^*$:} ligado \quad($H_u^L$: desligado).
\column{0.44\textwidth}
\centering
\includegraphics[width=\linewidth]{method1_saturation.png}\\
{\scriptsize Retornos decrescentes da qualidade vs.\ população}
\end{columns}
\bbfooter{18 genes nomeados $\;\to\;$ $m=18$ blocos vinculantes $\;\to\;$
$\Nhat_1=\mathbf{56}$ (pior gene 60)}
\end{frame}

\begin{frame}{M1 --- cálculo dos termos da fórmula}
\small
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Ordem $k_u=1$} (bloco de 1 gene) $\Rightarrow 2^{k_u-1}=2^0=1$.
\medskip

\textbf{$m$ (blocos vinculantes):} 18 nomeados
$\xrightarrow{\text{filtro signif.}}$ $m=18$ (nenhum descartado)
$\Rightarrow \sqrt{2m}=6{,}00$.
\medskip

\textbf{$\widehat\sigma_{BB,u},\widehat d_u$} ($R=60$ complementos,
$\Delta_u=F(\text{ON})-F(\text{OFF})$):
faixa $\widehat\sigma_{BB}\in[0{,}023,\,0{,}205]$,
$\widehat d\in[0{,}008,\,0{,}171]$.
\column{0.48\textwidth}
\textbf{Gene mais exigente} (define $\Nhat$): gene 60,
$\widehat d=0{,}0076$, $\widehat\sigma_{BB}=0{,}0233$.
\[
\widehat n=\frac{-\ln(0{,}05)\cdot 1\cdot \widehat\sigma_{BB}\sqrt{2m}}{\widehat d}
\]
\[
=\frac{2{,}996\cdot 0{,}0233\cdot 6{,}00}{0{,}0076}\approx \mathbf{56}
\]
\end{columns}
\end{frame}

% ---------------------------------------------------------------- M2
\begin{frame}{M2 --- Adjacência: BBs pela cobertura geométrica}
\begin{columns}[T]
\column{0.56\textwidth}
\small
\textbf{Sinal:} alcançabilidade. $A(t)[u,v]=1$ se $\lVert p_u-p_v\rVert\le R$;
cobertura $\cov(u)=\sum_t\sum_m A(t)[u,m]$.
\medskip

\textbf{Regra de BB:} candidatos \emph{indispensáveis por cobertura},
$\cov(u)\ge\theta TM$ ($\theta=0{,}10$) $\Rightarrow C_\theta=16$.
\medskip

\textbf{Interpretação:} relés que ficam em alcance de muitos pares
(móvel, instante) --- geometricamente necessários.
\medskip

\textbf{$H_u^*$:} ligado.
\column{0.44\textwidth}
\centering
\includegraphics[width=\linewidth]{method2_spatial.png}\\
{\scriptsize Candidatos coloridos por $\cov(u)$; estrela = sorvedouro}
\end{columns}
\bbfooter{16 genes nomeados $\;\to\;$ $m=10$ blocos vinculantes $\;\to\;$
$\Nhat_2=\mathbf{51}$ (pior gene 58)}
\end{frame}

\begin{frame}{M3 --- Roteamento: BBs pelo uso operacional}
\begin{columns}[T]
\column{0.56\textwidth}
\small
\textbf{Sinal:} uso real. BFS a partir do sorvedouro em cada $t$;
$\route(u)=\text{node\_acc}[u]/(TM)$ (fração de caminhos mínimos que usam $u$).

\medskip

\textbf{Interpretação:} relés que \emph{efetivamente carregam tráfego} (gargalos),
não apenas "em alcance". Mais esparso que a cobertura.

\section{Conclusão}
% ======================================================================
\begin{frame}{Conclusão}
\begin{itemize}
  \item A fórmula de ruína do jogador não estava errada --- estava
        \textbf{mal parametrizada} (ordem de bloco).
  \item \textbf{Bloco construtivo de P2 = ordem 1} (um gene indispensável por vez):
        $2^{k-1}=1$.
  \item Sob essa definição, a \emph{mesma} fórmula com \textbf{cinco fontes
        independentes} de BB concorda na faixa $\mathbf{26}$--$\mathbf{75}$.
  \item Recomendação: consenso $\mathbf{37}$, conservador $\mathbf{75}$
        (vs.\ $2839$ da aplicação ingênua).
\end{itemize}
